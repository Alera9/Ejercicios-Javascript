Description:

Write a function that performs the bubble algorithm. Input [3, 6, 12, 5, 100, 1] Output [1, 3, 5, 6, 12, 100]

Solution:
const bubbleSort = (nums) =>{
const arr = [];
nums.forEach(element => { arr.push(element*1); });
console.log(arr.length); let swapped = false;
for (let i = arr.length; i > 0; i--) {
swapped = true;
for (let j = 0; j < i-1; j++) {
if (arr[j] > arr[j+1]) {
[arr[j], arr[j+1]] = [arr[j+1], arr[j]] swapped = false;
}
} if(swapped) break;
} return arr;
}
